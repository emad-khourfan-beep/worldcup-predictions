function doGet(e) { if (e && e.parameter && e.parameter.action === "leaderboard") { return getLeaderboard(); } const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName("Matches"); const rows = sheet.getDataRange().getValues(); const matches = []; for (let i = 1; i < rows.length; i++) { matches.push({ id: rows[i][0], date: formatDate(rows[i][1]), time: formatTime(rows[i][2]), home: rows[i][3], away: rows[i][4] }); } return ContentService .createTextOutput(JSON.stringify(matches)) .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName("Prediction Details"); const data = JSON.parse(e.postData.contents); data.predictions.forEach(function(p) { sheet.appendRow([ new Date(), p.date || "", p.id, p.homeTeam + " × " + p.awayTeam, data.playerName, data.whatsapp || "", p.homeGoals + "-" + p.awayGoals, "Submitted", "", "", "" ]); }); return ContentService .createTextOutput("OK") .setMimeType(ContentService.MimeType.TEXT); } function calculateAllPoints() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName("Prediction Details"); const rows = sheet.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { const prediction = rows[i][6]; const actual = rows[i][8]; if (!prediction || !actual) continue; const p = prediction.toString().split("-").map(Number); const a = actual.toString().split("-").map(Number); let points = 0; let reason = ""; if (p[0] === a[0] && p[1] === a[1]) { points = 10; reason = "النتيجة الصحيحة بالكامل"; } else { const predictedResult = p[0] > p[1] ? "HOME" : p[0] < p[1] ? "AWAY" : "DRAW"; const actualResult = a[0] > a[1] ? "HOME" : a[0] < a[1] ? "AWAY" : "DRAW"; if (predictedResult === actualResult) { points += 5; reason = "الفائز أو التعادل صحيح"; } if (p[0] === a[0]) { points += 2; reason += reason ? " + أهداف الفريق الأول صحيحة" : "أهداف الفريق الأول صحيحة"; } if (p[1] === a[1]) { points += 2; reason += reason ? " + أهداف الفريق الثاني صحيحة" : "أهداف الفريق الثاني صحيحة"; } if (points === 0) { reason = "توقع خاطئ"; } } sheet.getRange(i + 1, 10).setValue(points); sheet.getRange(i + 1, 11).setValue(reason); } updateLeaderboard(); } function updateLeaderboard() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const detailsSheet = ss.getSheetByName("Prediction Details"); let leaderboardSheet = ss.getSheetByName("Leaderboard"); if (!leaderboardSheet) { leaderboardSheet = ss.insertSheet("Leaderboard"); } const rows = detailsSheet.getDataRange().getValues(); const players = {}; for (let i = 1; i < rows.length; i++) { const name = rows[i][4]; const points = Number(rows[i][9]) || 0; if (!name) continue; if (!players[name]) players[name] = 0; players[name] += points; } const ranking = Object.entries(players) .map(function(item) { return { name: item[0], points: item[1] }; }) .sort(function(a, b) { return b.points - a.points; }); leaderboardSheet.clear(); leaderboardSheet.appendRow(["Rank", "Name", "Total Points"]); ranking.forEach(function(player, index) { leaderboardSheet.appendRow([ index + 1, player.name, player.points ]); }); } function getLeaderboard() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName("Leaderboard"); if (!sheet) { return ContentService .createTextOutput(JSON.stringify([])) .setMimeType(ContentService.MimeType.JSON); } const rows = sheet.getDataRange().getValues(); const leaderboard = []; for (let i = 1; i < rows.length; i++) { leaderboard.push({ rank: rows[i][0], name: rows[i][1], points: rows[i][2] }); } return ContentService .createTextOutput(JSON.stringify(leaderboard)) .setMimeType(ContentService.MimeType.JSON); } function formatDate(value) { if (Object.prototype.toString.call(value) === "[object Date]") { return Utilities.formatDate(value, Session.getScriptTimeZone(), "yyyy-MM-dd"); } return value; } function formatTime(value) { if (Object.prototype.toString.call(value) === "[object Date]") { return Utilities.formatDate(value, Session.getScriptTimeZone(), "HH:mm"); } return value; }
